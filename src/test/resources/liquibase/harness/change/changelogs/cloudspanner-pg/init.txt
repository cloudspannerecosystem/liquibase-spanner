DROP TABLE IF EXISTS authors;

CREATE TABLE authors (
    id bigint NOT NULL,
    first_name varchar(1024),
    last_name varchar(1024),
    email varchar(1024),
    birthdate DATE,
    added timestamptz,
    PRIMARY KEY (id));

DROP TABLE IF EXISTS posts;

CREATE TABLE posts (
    id bigint NOT NULL,
    author_id bigint NOT NULL,
    title varchar(1024),
    description varchar(1024),
    content varchar(1024),
    inserted_date DATE,
    PRIMARY KEY (id));
